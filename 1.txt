package com.hubberspot.webservice;

import javax.jws.WebService;
import javax.jws.WebMethod;
import javax.jws.WebParam;

// @WebService annotation makes a class a Web Service
@WebService(serviceName = "CalculatorService")
public class CalculatorService {

    // @WebMethod annotation expose a method as a service.
    @WebMethod(operationName = "sum")
    // @WebParam annotation indicates parameters to method coming from SOAP request.
    public int sum(@WebParam(name = "number1") int number1, 
                   @WebParam(name = "number2") int number2) {
        
        int sum = 0;
        
        sum = number1 + number2;
        
        return sum;
    }    
}

import javax.xml.namespace.QName;
import javax.xml.ws.Service;
import java.net.URL;

public class CalculatorClient {

    public static void main(String[] args) throws Exception {
        // Replace with the actual endpoint URL of your web service
        String endpointURL = "http://localhost:8080/Calculator/CalculatorService"; 

        // Replace with the actual service name and port name
        String serviceName = "CalculatorService";
        String portName = "CalculatorServicePort";

        // Create a QName for the service and port
        QName serviceQName = new QName("http://webservice.hubberspot.com/", serviceName);
        QName portQName = new QName("http://webservice.hubberspot.com/", portName);

        // Create a Service object
        Service service = Service.create(serviceQName);

        // Create a port object
        URL wsdlURL = new URL(endpointURL + "?wsdl");
        CalculatorService port = service.getPort(portQName, CalculatorService.class);

        // Call the web service methods
        int sum = port.sum(5, 10);
        System.out.println("Sum: " + sum);
    }
}
